# 1.1 Discrepância de dados pode "mentir" porquê quando se tem 19 números "2", por exemplo, e um unico sendo o "162" sendo assim completando 20 números, ao tirar essa média se obtém 10 de média quando na realidade 95% se caracteriza como 2, logo pode se obter esse pensamento de ser "mentiroso".

# 1.2 Muitas vezes é mais viavél ignorar os valores extremos para que não ocorra os tão conhecidos outliers.

# 1.3 Margem de tolerância é utilizado em auditorias financeiras visando estabelecer valores mínimos e máximo para que eles analisem a solicitação com mais calma e frieza por ser um valor fora do comum vindo do cliente ou fora do comum para a própria a empresa mesmo.



,# 2.1. Configuração Global (Constante)

LIMITE_SEGURANCA = 10000.00



# 2.2. Processamento Interno (Função)

def analisar_vendas():

  # Usando a palavra-chave global para permitir alteração posterior

  global LIMITE_SEGURANCA



  print("--- INICIANDO AUDITORIA DE VENDAS ---")



  # 2.3. Entrada de Dados e Casting (Conversão para decimal)

  venda1 = float(input("Digite o valor da venda 1: R$ "))

  venda2 = float(input("Digite o valor da venda 2: R$ "))

  venda3 = float(input("Digite o valor da venda 3: R$ "))



  # 2.4. Cálculo da Média Simples

  media = (venda1 + venda2 + venda3) / 3



  print(f"\nMédia calculada: R$ {media:.2f}")



  # 2.5. Lógica de Investigação

  if media > LIMITE_SEGURANCA:

    print("⚠️ SISTEMA EM QUARENTENA")



    # Permitir que o gerente altere o limite se a venda for legítima

    confirmacao = input("As vendas são legítimas? Deseja alterar o LIMITE_SEGURANCA? (s/n): ")



    if confirmacao.lower() == 's':

      novo_limite = float(input("Digite o novo valor do limite: R$ "))

      LIMITE_SEGURANCA = novo_limite

      print(f"✅ Limite atualizado para: R$ {LIMITE_SEGURANCA:.2f}")

  else:

    print("✅ Vendas validadas com sucesso.")



# Execução do programa

if __name__ == "__main__":

  analisar_vendas()
