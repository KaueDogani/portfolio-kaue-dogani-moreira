def classificar_iqa(iqa):

  match iqa:

    case iqa if 0 <= iqa <= 40:

      return "Boa"

    case iqa if 41 <= iqa <= 80:

      return "Moderada"

    case iqa if 81 <= iqa <= 120:

      return "Ruim"

    case iqa if 121 <= iqa <= 200:

      return "Muito Ruim"

    case _:

      return "Péssima"





def calcular_conforto(temp, umidade, iqa):

  # Penalização de temperatura (ideal ~22°C)

  penal_temp = ((temp - 22) ** 2) / 50  



  # Penalização de umidade (ideal ~50%)

  penal_umidade = ((umidade - 50) ** 2) / 200  



  # Penalização de poluição (peso maior)

  penal_iqa = (iqa / 200) ** 2 * 5  



  # Nota base

  nota = 10 - (penal_temp + penal_umidade + penal_iqa)



  # Garantir intervalo 0–10

  return max(0, min(10, round(nota, 2)))





def main():

  # Lista de listas: [local, temperatura, umidade, iqa]

  dados_locais = [

    ["Praça do Forró", 29, 50, 70],

    ["Parque Ecológico Tietê", 28, 60, 55],

    ["Assaí Carrão", 30, 45, 80]

  ]



  for local in dados_locais:

    nome = local[0]

     

    print(f"\n📍 Local: {nome}")

     

    # Loop aninhado para métricas (temp, umidade, iqa)

    metricas = ["Temperatura", "Umidade", "IQA"]

     

    for i in range(1, len(local)):

      print(f"{metricas[i-1]}: {local[i]}")

     

    temp = local[1]

    umidade = local[2]

    iqa = local[3]



    classificacao = classificar_iqa(iqa)

    conforto = calcular_conforto(temp, umidade, iqa)



    print(f"Qualidade do Ar: {classificacao}")

    print(f"Nota de Conforto Urbano: {conforto}/10")





# Executar

main()



Parte 2: Navegação e Evacuação Espacial (avaliação + código 2)



1: lugar de uso foi a minha casa, e pela minha analise o único ponto de "saída de emergência" é pelo portão da frente ou portão principal. Tem 3 pontos de "nós" até a saída.



2: Meu quarto -> corredor (livre sem obstáculos) -> sala de estar (obstáculo um sofá que pode atrapalhar em uma emergência) -> quintal (possível obstáculo é o carro que pode estar estacionado) -> portão (saída).



CODIGO:

def simulador_evacucao():

  # Lista de locais (mapa linear)

  locais = [

    "Meu quarto",

    "Corredor",

    "Sala de estar",

    "Quintal",

    "Portão (Saída)"

  ]



  # Lista de estados/obstáculos correspondentes

  estados = [

    "chave disponível",     # quarto

    "livre",          # corredor

    "sofa bloqueando",     # sala

    "carro bloqueando",     # quintal

    "portao trancado"      # saída

  ]



  posicao = 0

  energia = 10 # limite de tentativas

  inventario = {

    "chave": False

  }



  print("🚨 Início da evacuação...\n")



  while energia > 0:

    local_atual = locais[posicao]

    estado_atual = estados[posicao]



    print(f"📍 Local atual: {local_atual}")

    print(f"🔎 Estado: {estado_atual}")

    print(f"⚡ Energia restante: {energia}")



    # IF ANINHADO para decisões

    if estado_atual == "chave disponível":

      print("🔑 Encontrou uma chave!")

      inventario["chave"] = True

      posicao += 1



    elif estado_atual == "livre":

      print("➡ Caminho livre, avançando...")

      posicao += 1



    elif estado_atual == "sofa bloqueando":

      print("🛋 Sofá no caminho...")

      if energia > 2:

        print("💪 Empurrando o sofá...")

        energia -= 2

        posicao += 1

      else:

        print("❌ Sem energia para mover o sofá, voltando...")

        posicao -= 1



    elif estado_atual == "carro bloqueando":

      print("🚗 Carro bloqueando passagem...")

      if inventario["chave"]:

        print("🔓 Usando chave para mover o carro!")

        posicao += 1

      else:

        print("❌ Não tem chave, voltando...")

        posicao -= 1



    elif estado_atual == "portao trancado":

      print("🚪 Portão trancado...")

      if inventario["chave"]:

        print("🎉 Usou a chave e escapou!")

        break

      else:

        print("❌ Sem chave! Voltando...")

        posicao -= 1



    # Evitar sair dos limites da lista

    if posicao < 0:

      posicao = 0

    elif posicao >= len(locais):

      posicao = len(locais) - 1



    print("-" * 40)

    energia -= 1



  if energia <= 0:

    print("\n💀 O agente ficou sem energia e falhou na evacuação.")

  else:

    print("\n🏃 Evacuação concluída com sucesso!")





# Executar

simulador_evacucao()



Parte 3: Reflexão critica (além do código)



Transformar um lugar real, como uma casa, em listas e regras para um programa faz a gente perceber que o mundo pode ser organizado de forma mais lógica. Coisas que antes pareciam confusas passam a ser vistas como situações com condições claras, tipo: “se tem obstáculo, então preciso fazer algo para passar”. Isso ajuda a tirar a bagunça das ideias e deixar tudo mais direto e objetivo.



Aprender programação também muda a forma de tomar decisões no dia a dia. A gente começa a pensar mais em “se isso acontecer, faço aquilo; senão, faço outra coisa”. Isso ajuda a planejar melhor, evitar erros e pensar antes de agir. Até tarefas simples, como organizar o dia, ficam mais fáceis quando você divide em etapas e pensa nas possibilidades.



Ao mesmo tempo, dá pra perceber que nem tudo na vida é tão exato quanto no código. Existem situações imprevisíveis e coisas que não seguem regras tão claras. Então, além de ajudar a pensar de forma lógica, programar também ensina que às vezes é preciso se adaptar. No fim, isso ajuda a resolver problemas de forma mais equilibrada e consciente.
